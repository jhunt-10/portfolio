def create_sequences(data, time_steps=6):
    X, y = [], []
    for i in range(time_steps, len(data)):
        X.append(data[i-time_steps:i, 0])  # Previous 'time_steps' data points
        y.append(data[i, 0])  # Predicting the 'close' price (1st column)

def create_sequences(data, time_steps=6):
    X, y = [], []
    for i in range(time_steps, len(data)):
        X.append(data[i-time_steps:i, 0])  # Using 'close' price only (1 column)
        y.append(data[i, 0])  # Target is also the 'close' price

def build_model(hp):
    model = Sequential()
    
    
    # Tune number of units in the first LSTM layer
    model.add(LSTM(units=hp.Int('units', min_value=32, max_value=512, step=32), 
                   return_sequences=True, 
                   input_shape=(6, 1)))  # Input shape is (time_steps, 1 feature)
    
    # Tune Dropout rate
    model.add(Dropout(rate=hp.Float('dropout_rate', min_value=0.1, max_value=0.5, step=0.1)))
    
    # Add another LSTM layer
    model.add(LSTM(units=hp.Int('units2', min_value=32, max_value=512, step=32), return_sequences=False))
    model.add(Dropout(rate=hp.Float('dropout_rate2', min_value=0.1, max_value=0.5, step=0.1)))
    
    # Dense layer
    model.add(Dense(units=hp.Int('dense_units', min_value=10, max_value=100, step=10)))
    
    # Output layer
    model.add(Dense(units=1))  # Output is one value (close price)
    
    # Tune learning rate for Adam optimizer
    learning_rate = hp.Float('lr', min_value=1e-5, max_value=1e-2, sampling='log')
    model.compile(optimizer=Adam(learning_rate=learning_rate), loss='mean_squared_error')
    
    return model
